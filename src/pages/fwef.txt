// ================= AUDIO ENGINE =================
class AudioEngine {
  ctx: AudioContext;
  masterGain: GainNode;
  reverbNode: ConvolverNode;
  delayNode: DelayNode;
  delayGain: GainNode;
  analyzer: AnalyserNode;
  samples: Map<string, AudioBuffer> = new Map();

  constructor() {
    this.ctx = new (window.AudioContext || (window as any).webkitAudioContext)();
    this.masterGain = this.ctx.createGain();
    this.analyzer = this.ctx.createAnalyser();
    
    // FX Chains
    this.delayNode = this.ctx.createDelay(2.0);
    this.delayGain = this.ctx.createGain();
    this.delayNode.delayTime.value = 0.375;
    this.delayGain.gain.value = 0.3;
    
    // Connect Chain
    this.masterGain.connect(this.analyzer);
    this.analyzer.connect(this.ctx.destination);
    
    this.delayNode.connect(this.delayGain);
    this.delayGain.connect(this.masterGain);
    this.delayGain.connect(this.delayNode);

    this.masterGain.gain.value = 0.7;
    this.analyzer.fftSize = 256;
  }

  async loadSample(file: File): Promise<string> {
    const arrayBuffer = await file.arrayBuffer();
    const audioBuffer = await this.ctx.decodeAudioData(arrayBuffer);
    const id = crypto.randomUUID();
    this.samples.set(id, audioBuffer);
    return id;
  }

  playMetronome(time: number, isDownbeat: boolean) {
    const now = this.ctx.currentTime + time;
    const osc = this.ctx.createOscillator();
    const g = this.ctx.createGain();
    osc.frequency.setValueAtTime(isDownbeat ? 1000 : 600, now);
    g.gain.setValueAtTime(0.2, now);
    g.gain.exponentialRampToValueAtTime(0.001, now + 0.05);
    osc.connect(g);
    g.connect(this.masterGain);
    osc.start(now);
    osc.stop(now + 0.1);
  }

  playNote(track: Track, pitch: number, time: number, duration: number, velocity: number = 0.8) {
    const now = this.ctx.currentTime + time;
    const freq = 440 * Math.pow(2, (pitch - 69) / 12);
    const vol = velocity * track.volume;

    const g = this.ctx.createGain();
    g.gain.setValueAtTime(0, now);
    g.gain.linearRampToValueAtTime(vol * 0.4, now + 0.01);
    g.gain.exponentialRampToValueAtTime(0.001, now + duration);

    if (track.customSample && this.samples.has(track.customSample)) {
      const source = this.ctx.createBufferSource();
      source.buffer = this.samples.get(track.customSample)!;
      source.connect(g);
      source.start(now);
      source.stop(now + duration + 0.1);
    } else {
      const osc = this.ctx.createOscillator();
      osc.frequency.setValueAtTime(freq, now);
      osc.type = track.instrument === 'synth' ? 'sawtooth' : track.instrument === 'bass' ? 'triangle' : 'sine';
      osc.connect(g);
      osc.start(now);
      osc.stop(now + duration + 0.1);
    }

    g.connect(this.masterGain);
    
    if (track.delaySend > 0) {
      const fxGain = this.ctx.createGain();
      fxGain.gain.value = track.delaySend;
      g.connect(fxGain);
      fxGain.connect(this.delayNode);
    }
  }
}

// ================= APP COMPONENT =================
export default function App() {
  const [project, setProject] = useState<MIDIProject | null>(null);
  const [zoom, setZoom] = useState(180);
  const [selectedNotes, setSelectedNotes] = useState<string[]>([]);
  const [activeNotes, setActiveNotes] = useState<Set<number>>(new Set());
  const [dragging, setDragging] = useState<{ id: string; type: 'move' | 'resize'; startX: number; origStart: number; origPitch: number; origDur: number } | null>(null);
  
  const engineRef = useRef<AudioEngine | null>(null);
  const rafRef = useRef<number | null>(null);
  const lastTimeRef = useRef(0);
  const lastMetronomeBeatRef = useRef(-1);

  const getEngine = useCallback(() => {
    if (!engineRef.current) engineRef.current = new AudioEngine();
    if (engineRef.current.ctx.state === 'suspended') engineRef.current.ctx.resume();
    return engineRef.current;
  }, []);

  // Save/Load
  useEffect(() => {
    const saved = localStorage.getItem('aura_pro_sequencer');
    if (saved) try { setProject(JSON.parse(saved)); } catch {}
  }, []);

  useEffect(() => {
    if (project) localStorage.setItem('aura_pro_sequencer', JSON.stringify(project));
  }, [project]);

  const createNewProject = () => {
    const trackId = crypto.randomUUID();
    setProject({
      id: crypto.randomUUID(),
      name: "midnight_studio_v1",
      bpm: 124,
      tracks: [{
        id: trackId,
        name: "Neon Lead",
        instrument: 'synth',
        color: '#00D1FF',
        notes: [],
        muted: false, solo: false, volume: 0.8, pan: 0,
        reverbSend: 0, delaySend: 0.2
      }],
      currentTime: 0,
      isPlaying: false,
      activeTrackId: trackId,
      patternLength: 16,
      metronomeEnabled: true
    });
  };

  const togglePlayback = useCallback(() => {
    if (project?.isPlaying) {
      stopPlayback();
    } else {
      startPlayback();
    }
  }, [project]);

  const startPlayback = useCallback(() => {
    const eng = getEngine();
    if (!project) return;
    
    setProject(prev => prev ? { ...prev, isPlaying: true } : null);
    lastTimeRef.current = eng.ctx.currentTime;
    lastMetronomeBeatRef.current = -1;

    const tick = () => {
      const now = eng.ctx.currentTime;
      const delta = now - lastTimeRef.current;
      lastTimeRef.current = now;

      setProject(prev => {
        if (!prev || !prev.isPlaying) return prev;
        
        const bps = prev.bpm / 60;
        const beatDelta = delta * bps;
        const newTime = (prev.currentTime + beatDelta) % prev.patternLength;

        const currentBeat = Math.floor(prev.currentTime);
        if (prev.metronomeEnabled && currentBeat !== lastMetronomeBeatRef.current) {
          eng.playMetronome(0, currentBeat % 4 === 0);
          lastMetronomeBeatRef.current = currentBeat;
        }

        const lookahead = 0.1;
        const lookaheadBeats = lookahead * bps;

        prev.tracks.forEach(track => {
          if (track.muted) return;
          const soloExists = prev.tracks.some(t => t.solo);
          if (soloExists && !track.solo) return;

          track.notes.forEach(note => {
            if (note.startTime >= prev.currentTime && note.startTime < prev.currentTime + lookaheadBeats) {
              const delay = (note.startTime - prev.currentTime) / bps;
              eng.playNote(track, note.pitch, delay, note.duration / bps, note.velocity / 127);
            }
          });
        });

        return { ...prev, currentTime: newTime };
      });
      rafRef.current = requestAnimationFrame(tick);
    };
    rafRef.current = requestAnimationFrame(tick);
  }, [project, getEngine]);

  const stopPlayback = useCallback(() => {
    if (rafRef.current) cancelAnimationFrame(rafRef.current);
    setProject(prev => prev ? { ...prev, isPlaying: false, currentTime: 0 } : null);
  }, []);

  const handleGridClick = (e: React.MouseEvent) => {
    if (!project || !project.activeTrackId) return;
    const rect = e.currentTarget.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;

    const beat = x / zoom;
    const pitch = HIGHEST_NOTE - Math.floor(y / NOTE_HEIGHT);
    const quantizedBeat = Math.floor(beat * 8) / 8;

    setProject(prev => {
      if (!prev) return prev;
      const track = prev.tracks.find(t => t.id === prev.activeTrackId);
      if (!track) return prev;

      const existingIdx = track.notes.findIndex(n => n.pitch === pitch && Math.abs(n.startTime - quantizedBeat) < 0.05);
      
      if (existingIdx !== -1) {
        const newNotes = [...track.notes];
        newNotes.splice(existingIdx, 1);
        return { ...prev, tracks: prev.tracks.map(t => t.id === prev.activeTrackId ? { ...t, notes: newNotes } : t) };
      }

      const newNote: Note = {
        id: crypto.randomUUID(),
        pitch,
        startTime: quantizedBeat,
        duration: 0.25,
        velocity: 100,
        trackId: prev.activeTrackId
      };
      
      getEngine().playNote(track, pitch, 0, 0.2);
      return { ...prev, tracks: prev.tracks.map(t => t.id === prev.activeTrackId ? { ...t, notes: [...t.notes, newNote] } : t) };
    });
  };

  const handleFileUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file || !project) return;
    const eng = getEngine();
    const sid = await eng.loadSample(file);
    const tid = crypto.randomUUID();
    const newTrack: Track = {
      id: tid,
      name: file.name.split('.')[0],
      instrument: 'custom',
      color: '#FFFFFF',
      notes: [],
      muted: false, solo: false, volume: 0.8, pan: 0,
      customSample: sid,
      reverbSend: 0, delaySend: 0
    };
    setProject({ ...project, tracks: [...project.tracks, newTrack], activeTrackId: tid });
  };

  const addDefaultTrack = (type: InstrumentType) => {
    if (!project) return;
    const inst = INSTRUMENTS.find(i => i.id === type) || INSTRUMENTS[0];
    const tid = crypto.randomUUID();
    setProject({ 
      ...project, 
      tracks: [...project.tracks, {
        id: tid,
        name: inst.name, instrument: type, color: inst.color, notes: [],
        muted: false, solo: false, volume: 0.7, pan: 0, reverbSend: 0, delaySend: 0
      }],
      activeTrackId: tid
    });
  };

  if (!project) {
    return (
      <div className="midi-root" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', justifyContent: 'center', minHeight: '100vh', background: '#010101' }}>
        <div style={{ textAlign: 'center' }}>
          <div style={{ width: 80, height: 80, background: '#00D1FF', borderRadius: '2rem', margin: '0 auto 32px', display: 'flex', alignItems: 'center', justifyContent: 'center', boxShadow: '0 0 30px rgba(0,209,255,0.3)' }}>
            <Speaker style={{ color: 'white', width: 40, height: 40 }} />
          </div>
          <h1 style={{ fontSize: '48px', fontWeight: 'bold', marginBottom: '8px' }}>NEON STUDIO</h1>
          <p style={{ color: '#888', marginBottom: '40px', fontSize: '12px', textTransform: 'uppercase', letterSpacing: '2px' }}>High Fidelity Sequencer</p>
          <button onClick={createNewProject} style={{ padding: '12px 32px', background: 'white', color: 'black', fontWeight: 'bold', borderRadius: '16px', border: 'none', cursor: 'pointer' }}>New Session</button>
        </div>
      </div>
    );
  }

  return (
    <div style={{ display: 'flex', flexDirection: 'column', height: '100vh', background: '#010101', color: '#F5F5F7' }}>
      {/* Dynamic Header */}
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', padding: '0 24px', height: 56, borderBottom: '1px solid rgba(255,255,255,0.05)', background: 'rgba(0,0,0,0.4)' }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 40 }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>
            <div style={{ width: 32, height: 32, borderRadius: 8, background: '#00D1FF', display: 'flex', alignItems: 'center', justifyContent: 'center' }}><Activity size={16} style={{ color: 'black' }} /></div>
            <span style={{ fontSize: '10px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '1px' }}>{project.name}</span>
          </div>
          <div style={{ display: 'flex', alignItems: 'center', gap: 32 }}>
            <div style={{ display: 'flex', borderLeft: '1px solid rgba(255,255,255,0.05)', paddingLeft: 32, flexDirection: 'column' }}>
              <span style={{ fontSize: '9px', color: '#666', fontWeight: 'bold', textTransform: 'uppercase' }}>Tempo</span>
              <input type="number" value={project.bpm} onChange={e => setProject({...project, bpm: +e.target.value})} style={{ background: 'transparent', fontSize: '14px', fontFamily: 'monospace', width: 48, border: 'none', color: 'white', outline: 'none' }} />
            </div>
            <div style={{ display: 'flex', borderLeft: '1px solid rgba(255,255,255,0.05)', paddingLeft: 32, flexDirection: 'column' }}>
              <span style={{ fontSize: '9px', color: '#666', fontWeight: 'bold', textTransform: 'uppercase' }}>Metronome</span>
              <button 
                onClick={() => setProject({...project, metronomeEnabled: !project.metronomeEnabled})}
                style={{ fontSize: '9px', fontWeight: 'bold', background: 'none', border: 'none', cursor: 'pointer', color: project.metronomeEnabled ? '#00D1FF' : '#666' }}
              >
                {project.metronomeEnabled ? 'ON' : 'OFF'}
              </button>
            </div>
          </div>
        </div>

        <div style={{ display: 'flex', alignItems: 'center', gap: 16 }}>
          <div style={{ display: 'flex', background: 'rgba(255,255,255,0.05)', borderRadius: 12, padding: 4, border: '1px solid rgba(255,255,255,0.05)' }}>
            <button onClick={stopPlayback} style={{ padding: '8px', background: 'none', border: 'none', borderRadius: 8, cursor: 'pointer', color: '#999' }}><Square size={14} /></button>
            <button onClick={togglePlayback} style={{ padding: '8px', background: 'none', border: 'none', borderRadius: 8, cursor: 'pointer', color: '#00D1FF' }}>
              {project.isPlaying ? <Pause size={16} /> : <Play size={16} />}
            </button>
          </div>
          <button onClick={() => setProject(null)} style={{ padding: '8px', background: 'none', border: 'none', color: '#666', cursor: 'pointer' }}><X size={18} /></button>
        </div>
      </div>

      {/* Main Studio View */}
      <div style={{ flex: 1, display: 'flex', overflow: 'hidden' }}>
        {/* Layer Manager */}
        <div style={{ width: 288, borderRight: '1px solid rgba(255,255,255,0.05)', background: 'rgba(0,0,0,0.4)', display: 'flex', flexDirection: 'column', flexShrink: 0 }}>
          <div style={{ padding: 20, display: 'flex', flexDirection: 'column', gap: 24, overflowY: 'auto' }}>
            <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
              <h2 style={{ fontSize: '10px', textTransform: 'uppercase', fontWeight: 'bold', letterSpacing: '1px', color: '#666' }}>Track List</h2>
              <div style={{ display: 'flex', gap: 8 }}>
                <input type="file" id="sample-in" style={{ display: 'none' }} accept="audio/*" onChange={handleFileUpload} />
                <label htmlFor="sample-in" style={{ padding: '6px', borderRadius: 8, cursor: 'pointer' }}><Upload size={14} /></label>
              </div>
            </div>

            <div style={{ display: 'flex', flexDirection: 'column', gap: 8 }}>
              {project.tracks.map(track => (
                <div key={track.id} onClick={() => setProject({...project, activeTrackId: track.id})} style={{ padding: 16, borderRadius: 16, border: '1px solid', cursor: 'pointer', transition: 'all 0.2s', background: project.activeTrackId === track.id ? 'rgba(255,255,255,0.1)' : 'transparent', borderColor: project.activeTrackId === track.id ? 'rgba(255,255,255,0.2)' : 'transparent', opacity: project.activeTrackId === track.id ? 1 : 0.4 }}>
                  <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 8 }}>
                    <span style={{ fontSize: '11px', fontWeight: 'bold', overflow: 'hidden', textOverflow: 'ellipsis', maxWidth: 120 }}>{track.name}</span>
                    <div style={{ display: 'flex', gap: 4 }}>
                      <button onClick={e => {e.stopPropagation(); setProject({...project, tracks: project.tracks.map(t => t.id === track.id ? {...t, muted: !t.muted} : t)})}} style={{ width: 20, height: 20, display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: '8px', fontWeight: 'bold', borderRadius: 6, border: 'none', cursor: 'pointer', background: track.muted ? '#ff4444' : 'rgba(255,255,255,0.1)', color: 'white' }}>M</button>
                      <button onClick={e => {e.stopPropagation(); setProject({...project, tracks: project.tracks.map(t => t.id === track.id ? {...t, solo: !t.solo} : t)})}} style={{ width: 20, height: 20, display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: '8px', fontWeight: 'bold', borderRadius: 6, border: 'none', cursor: 'pointer', background: track.solo ? '#00D1FF' : 'rgba(255,255,255,0.1)', color: 'white' }}>S</button>
                    </div>
                  </div>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
                    <Volume2 size={10} style={{ color: '#666' }} />
                    <div style={{ flex: 1, height: 2, background: 'rgba(255,255,255,0.1)', borderRadius: 2 }}>
                      <div style={{ height: '100%', background: '#999', width: `${track.volume * 100}%`, borderRadius: 2 }} />
                    </div>
                  </div>
                </div>
              ))}
            </div>

            <div>
              <h3 style={{ fontSize: '10px', textTransform: 'uppercase', fontWeight: 'bold', letterSpacing: '1px', color: '#666', marginBottom: 16 }}>Presets</h3>
              <div style={{ display: 'grid', gridTemplateColumns: 'repeat(2, 1fr)', gap: 8 }}>
                {INSTRUMENTS.map(inst => (
                  <button key={inst.id} onClick={() => addDefaultTrack(inst.id)} style={{ padding: 12, background: 'rgba(255,255,255,0.03)', border: '1px solid rgba(255,255,255,0.05)', borderRadius: 12, cursor: 'pointer', transition: 'all 0.2s', display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 8 }}>
                    <inst.icon size={14} style={{ color: '#666' }} />
                    <span style={{ fontSize: '8px', fontWeight: 'bold', letterSpacing: '1px' }}>{inst.name}</span>
                  </button>
                ))}
              </div>
            </div>
          </div>
        </div>

            <span>TIMELINE VIEW</span>
            <div style={{ display: 'flex', gap: 16 }}>
              <button onClick={() => setZoom(z => Math.max(50, z - 20))} style={{ background: 'none', border: 'none', color: '#666', cursor: 'pointer' }}>Zoom -</button>
              <button onClick={() => setZoom(z => Math.min(600, z + 20))} style={{ background: 'none', border: 'none', color: '#666', cursor: 'pointer' }}>Zoom +</button>
              <button onClick={() => setProject({...project, tracks: project.tracks.map(t => ({...t, notes: []}))})} style={{ background: 'none', border: 'none', color: '#ff4444', cursor: 'pointer', fontWeight: 'bold' }}>Clear All</button>
            </div>
          </div>

          <div style={{ flex: 1, overflow: 'auto', position: 'relative', userSelect: 'none' }}>
            <div style={{ display: 'flex', minHeight: '100%' }}>
              {/* Vertical Keyboard */}

                {Array.from({ length: TOTAL_NOTES }).map((_, i) => {
                  const pitch = HIGHEST_NOTE - i;
                  const isC = pitch % 12 === 0;
                  const isBlack = [1, 3, 6, 8, 10].includes(pitch % 12);
                  return (

'rgba(0,0,0,0.4)' : 'transparent', color: isC ? '#00D1FF' : '#666', fontWeight: isC ? 'bold' : 'normal' }}>

                {/* Grid Guides */}
                {Array.from({ length: TOTAL_NOTES }).map((_, i) => (
                  <div key={i} style={{ position: 'absolute', left: 0, right: 0, height: 1, background: 'rgba(255,255,255,0.03)', top: i * NOTE_HEIGHT }} />
